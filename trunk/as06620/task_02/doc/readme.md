                    - 0.01 * 19.5 * 19.5
                    + 0.3 * 10.0
                    + 0.1 * sin(5.0);

TEST(TemperatureModelTest, NonlinearMultipleSteps) {
    TemperatureModel model(0.7, 0.01, 0.3, 0.1);
