## Ход работы

### Пример тестов
```cpp
TEST(LinearModel, ZeroCase) {
    EXPECT_DOUBLE_EQ(linear(0, 0), 0);
}
TEST(LinearModel, OnlyY) {
    EXPECT_DOUBLE_EQ(linear(18, 0), a * 18);
}
TEST(LinearModel, OnlyU) {
    EXPECT_DOUBLE_EQ(linear(0, 5), b * 5);
}
TEST(LinearModel, MixedInput) {
    EXPECT_DOUBLE_EQ(linear(18, 5), a * 18 + b * 5);
}

TEST(NonlinearModel, ZeroCase) {
    EXPECT_DOUBLE_EQ(nonlinear(0, 0, 0, 0), 0);
}
TEST(NonlinearModel, OnlyY) {
    EXPECT_DOUBLE_EQ(nonlinear(18, 18, 0, 0), a * 18 - b * pow(18, 2));
}
TEST(NonlinearModel, OnlyU) {
    EXPECT_DOUBLE_EQ(nonlinear(0, 0, 5, 5), c * 5 + d * std::sin(5));
}
TEST(NonlinearModel, MixedInput) {
    EXPECT_DOUBLE_EQ(nonlinear(18, 18, 5, 5),
                     a * 18 - b * pow(18, 2) + c * 5 + d * std::sin(5));
}
